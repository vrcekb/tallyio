#[cfg(test)]
mod tests {
    use super::*;
    use std::time::Instant;

    #[test]
    fn test_CRATE_manager_creation() -> CRATEResult<()> {
        let manager = CRATEManager::new()?;
        assert_eq!(manager.COUNT.load(std::sync::atomic::Ordering::Relaxed), 0);
        Ok(())
    }

    #[test]
    fn test_CRATE_manager_default() {
        let manager = CRATEManager::default();
        assert_eq!(manager.COUNT.load(std::sync::atomic::Ordering::Relaxed), 0);
    }

    #[test]
    fn test_METHOD() -> CRATEResult<()> {
        let manager = CRATEManager::new()?;
        let result = manager.METHOD("test_data")?;
        
        // Verify operation was processed
        assert_eq!(result, "RESULT test_data");
        assert_eq!(manager.COUNT.load(std::sync::atomic::Ordering::Relaxed), 1);
        Ok(())
    }

    #[test]
    fn test_CRATE_latency_requirement() -> CRATEResult<()> {
        let manager = CRATEManager::new()?;
        let start = Instant::now();
        
        manager.METHOD("latency_test")?;
        
        let duration = start.elapsed();
        assert!(duration.as_millis() < 1, "CRATE operation took {}ms, must be <1ms", duration.as_millis());
        Ok(())
    }

    #[test]
    fn test_multiple_operations() -> CRATEResult<()> {
        let manager = CRATEManager::new()?;
        
        for i in 0..10 {
            manager.METHOD(&format!("operation_{}", i))?;
        }
        
        assert_eq!(manager.COUNT.load(std::sync::atomic::Ordering::Relaxed), 10);
        Ok(())
    }
}
